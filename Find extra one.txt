#A - Find Extra One -> (REFERENCE TO CODEFORCES)

You have n distinct points on a plane, none of them lie on OY axis. Check that there is a point after removal of which the remaining points are located on one side of the OY axis.

#LOGIC ->

	Logic of this problem is simple as you need to find if there any point which after removing all points are equally divided by Y-axis. If all points are same side of Y-axis, then remove anyone point it won't effect.

#CODE ->

	#include<iostream>
	#include<algorithm>
 
	using namespace std;
	int main()
	{
    	long n,p=0,ng=0;
    	cin>>n;
    	long long int x[n],y[n];
    	for(long i=0;i<n;i++)
        	cin>>x[i]>>y[i];
    	if(n==2)
        	cout<<"YES"<<endl;
    	else
    	{
        	for( long i=0;i<n;i++)
        	{
            		if(x[i]>0)
            		{
                		p++;
            		}
            		else if(x[i]<0)
            		{
                	ng++;
            		}
        	}
        	if(p==ng)
            		cout<<"NO"<<endl;
        	else if(ng==1 || p==1 || ng==0 || p==0)
            		cout<<"YES"<<endl;
        	else
            		cout<<"NO"<<endl;
    	}
    	return 0;
	}
